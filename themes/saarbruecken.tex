% create custom styles
\tikzset{
  title/.style={
  drop shadow,
  draw=black,
    opacity=1.0,
    text opacity=1.0,
    fill=brown!20,
    rounded corners
  },
  posterblock/.style={
  drop shadow,
  draw=black,
    opacity=1.0,
    text opacity=1.0,
    fill=brown!20,
    rounded corners
  },
  logobar/.style={
    % make the bar a little bit wider than the paper width,
    % such that the left and right borders are not visible
    minimum width=\paperwidth + 10pt,
    text width=\textwidth,
    fill=white,
    draw=black!50,
    very thick,
    inner sep=5
  }
}

% tweak layout
\setmargins{15}{15}
\setblockpadding{20}
\setblockspacing{15}
\shadebackground{blue!10}{blue!10}

    \begin{posterblock}[title]
        \begin{center}
            {\huge{\bf{\@title}}} \\
            \vspace{5pt}
            {\large\@author}
        \end{center}

    \end{posterblock}
}
\makeatother
